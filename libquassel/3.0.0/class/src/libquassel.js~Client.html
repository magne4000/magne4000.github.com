    Client

    Extends:

    EventEmitter → Client

    Main class to interact with Quassel instance

    Example:

    import { Client } from 'libquassel';
const net = require('net');

const quassel = new Client(function(next) {
  next("user", "password");
});
const socket = net.createConnection({
  host: "localhost",
  port: 4242
});

quassel.connect(socket);
